    ```typescript
    import { z } from "zod";

    // Global Settings -- stored securely, NOT exported in profiles (CONN-06)
    export const GlobalSettingsSchema = z.object({
      broker: z.object({
        host: z.string().min(1),
        port: z.number().int().min(1).max(65535).default(1883),
        username: z.string().optional(),
        password: z.string().optional(),
        tls: z.boolean().default(false),
        clientId: z.string().optional(), // auto-generated if missing
      }).optional(),
    });
    export type GlobalSettings = z.infer<typeof GlobalSettingsSchema>;

    // Broker config extracted for ConnectionManager use
    export interface BrokerConfig {
      host: string;
      port: number;
      username?: string;
      password?: string;
      tls: boolean;
      clientId?: string;
    }

    // Action Settings -- per button, exported in profiles (no secrets!)
    export const MqttActionSettingsSchema = z.object({
      subscribeTopic: z.string().optional(),
      publishTopic: z.string().optional(),
      publishPayload: z.string().optional(),
      qos: z.union([z.literal(0), z.literal(1), z.literal(2)]).default(0),
      retain: z.boolean().default(false),
      lastValue: z.string().optional(), // cached for restart recovery
    });
    export type MqttActionSettings = z.infer<typeof MqttActionSettingsSchema>;
    ```

    **1. src/services/connection-manager.ts** -- Singleton managing MQTT connections. One TCP connection per unique broker (CONN-05, ARCH-02).

    Key implementation details:
    - Private `clients: Map&lt;string, mqtt.MqttClient&gt;` keyed by `brokerKey()` output
    - Private `activeTopics: Map&lt;string, Set&lt;string&gt;&gt;` -- broker key to set of subscribed topics
    - `getOrCreate(config: BrokerConfig): mqtt.MqttClient` -- creates client if not exists, returns existing otherwise
    - On client creation, use `mqtt.connect()` with these options:
      - `protocol: config.tls ? "mqtts" : "mqtt"` (CONN-01, CONN-03)
      - `host: config.host`, `port: config.port` (CONN-01)
      - `username: config.username || undefined`, `password: config.password || undefined` (CONN-02)
      - `reconnectPeriod: 5000` (CONN-04)
      - `clean: true` (per Pitfall 1 -- explicit resubscribe is more reliable)
      - `clientId: config.clientId || "streamdeck-mqtt-" + crypto.randomUUID()` (per Pitfall 4)
      - `rejectUnauthorized: false` for TLS (CONN-03 -- self-signed certs common in home networks)
    - Register `client.on("connect", ...)` handler that resubscribes ALL topics from `activeTopics` for this broker key (Pitfall 1 fix: do NOT rely on MQTT.js resubscribe)
    - Register `client.on("message", (topic, payload) => ...)` handler that calls `topicRouter.dispatch(key, topic, payload.toString())`
    - Register `client.on("error", ...)` and `client.on("offline", ...)` with logger output
    - `ensureSubscribed(brokerKeyStr: string, topic: string): void` -- adds topic to activeTopics set, calls `client.subscribe(topic)` if client is connected
    - `ensureUnsubscribed(brokerKeyStr: string, topic: string): void` -- removes topic from activeTopics, calls `client.unsubscribe(topic)` if client is connected
    - `reset(): void` -- clears all tracking state (for Pitfall 2 -- plugin restart recovery)
    - Export a singleton instance: `export const connectionManager = new ConnectionManager()`

    Import `topicRouter` from `./topic-router` for the message dispatch.

    **2. src/services/topic-router.ts** -- Routes MQTT messages to button action contexts (ARCH-03).

    Key implementation details:
    - Private `subscriptions: Map&lt;string, Map&lt;string, Set&lt;string&gt;&gt;&gt;` -- brokerKey -> topic -> Set of action context IDs
    - Private `callbacks: Map&lt;string, (payload: string) => void&gt;` -- contextId -> callback function
    - `register(brokerKey: string, topic: string, contextId: string, callback: (payload: string) => void): boolean` -- returns `true` if this is the first subscriber for this topic on this broker (caller should MQTT subscribe)
    - `unregister(brokerKey: string, topic: string, contextId: string): boolean` -- returns `true` if this was the last subscriber (caller should MQTT unsubscribe)
    - `dispatch(brokerKey: string, topic: string, payload: string): void` -- iterates all contextIds for this broker+topic, calls their registered callbacks
    - `reset(): void` -- clears all subscriptions and callbacks (Pitfall 2 fix)
    - `getTopics(brokerKey: string): string[]` -- returns all topics for a broker (for resubscribe on connect)
    - Export singleton: `export const topicRouter = new TopicRouter()`
